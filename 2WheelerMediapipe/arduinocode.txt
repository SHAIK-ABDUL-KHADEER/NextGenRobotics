#include <ESP8266WiFi.h>
#include <WiFiUdp.h>

// ---------- AP Details ----------
const char* ssid = "2Wheeler";
const char* password = "12345678";

// ---------- UDP ----------
WiFiUDP udp;
unsigned int localUdpPort = 4210;
char incomingPacket[255];

// ---------- Motor Pins ----------
#define IN1 D0
#define IN2 D1
#define IN3 D2
#define IN4 D3
#define ENA D5
#define ENB D6

// ---------- Motor Speed ----------
int motorSpeed = 800;

// ---------- Motor Functions ----------
void stopMotor() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);
}

void moveForward() {
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
}

void moveBackward() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, HIGH);
}

void turnLeft() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
}

void turnRight() {
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, HIGH);
}

// Optional Dance Mode
void dance() {
  turnLeft(); delay(300);
  turnRight(); delay(300);
  moveForward(); delay(300);
  moveBackward(); delay(300);
  stopMotor();
}

// ---------- Setup ----------
void setup() {
  Serial.begin(9600);

  // Motor setup
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);
  pinMode(ENA, OUTPUT);
  pinMode(ENB, OUTPUT);

  analogWrite(ENA, motorSpeed);
  analogWrite(ENB, motorSpeed);

  stopMotor();

  // âœ… AP Mode Start
  WiFi.mode(WIFI_AP);
  WiFi.softAP(ssid, password);

  Serial.println("\nâœ… NodeMCU AP Mode Started!");
  Serial.print("ðŸ“¡ WiFi Name: ");
  Serial.println(ssid);
  Serial.print("ðŸ” Password: ");
  Serial.println(password);

  Serial.print("ðŸ“ AP IP Address: ");
  Serial.println(WiFi.softAPIP());  // usually 192.168.4.1

  // âœ… Start UDP listening
  udp.begin(localUdpPort);
  Serial.print("âœ… UDP Listening on Port: ");
  Serial.println(localUdpPort);
}

// ---------- Loop ----------
void loop() {
  int packetSize = udp.parsePacket();

  if (packetSize) {
    int len = udp.read(incomingPacket, 255);
    if (len > 0) incomingPacket[len] = 0;

    String cmd = String(incomingPacket);
    cmd.trim();

    Serial.print("ðŸ“© Received UDP Command: ");
    Serial.println(cmd);

    // âœ… Command Handling
    if (cmd == "FORWARD") {
      moveForward();
    }
    else if (cmd == "BACKWARD") {
      moveBackward();
    }
    else if (cmd == "LEFT") {
      turnLeft();
    }
    else if (cmd == "RIGHT") {
      turnRight();
    }
    else if (cmd == "STOP") {
      stopMotor();
    }
    else if (cmd == "DANCE") {
      dance();
    }
    else {
      Serial.println("âš  Unknown Command!");
      stopMotor();
    }
  }
}
